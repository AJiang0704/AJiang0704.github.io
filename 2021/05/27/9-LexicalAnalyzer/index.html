本实验要求学生编制一个读单词过程，从输入的源程序中，识别出各个具有独立意义的单词，即基本保留字、标识符、常数、运算符、分隔符五大类。并依次输出各个单词的内部编码及单词符号自身值。（遇到错误时可显示“Error”，然后跳过错误部分继续进行）。

LexicalAnalyzer.cpp

#include
#include
#include

int t = -1;
int pProject = 0;

/*****保留字*****/
static char reserveWord[32][10] = {
    "auto", "break", "case", "char", "const", "continue", 
    "default", "do", "double", "else", "enum", "extern", 
    "float", "for", "goto", "if", "int", "long", "register", 
    "return", "short", "signed", "sizeof", "static", 
    "struct", "switch", "typedef", "union", "unsigned", 
    "void", "volatile", "while"
};

/*****运算符*****/
static char operatorWord[22][10] = {
    "+", "-", "*", "/", "%", "++","--","<", "<=", ">", ">=", 
    "=", "==", "!=", "&", "&&", "|", "||", "!", "^","<<", ">>"
};

/*****分隔符*****/
static char boredrWord[16][10] = {
    "(", ")", "[", "]", "{", "}", ";", ",", "\"", "\'", 
    "#", "~", "\\", ".", "\?", ":"
};

/*****标识符*****/
static  char identifier[1000][50] = {""};

/*****过滤源程序函数*****/
int filtrationResource( char resourceProject[] , int end ) {
    char temp[10000]; 
    int count = 0;
    int i;
    for( i = 0 ; i < end ; i++ ) {
        //过滤头文件 
        if ( resourceProject[i] == '#' && resourceProject[i + 1] == 'i' && resourceProject[i + 2] == 'n' && resourceProject[i + 3] == 'c' 
        && resourceProject[i + 4] == 'l' && resourceProject[i + 5] == 'u' && resourceProject[i + 6] == 'd' && resourceProject[i + 7] == 'e' ) {
            printf("(  头文件  ,  ");
            while( resourceProject[i] !='<') {
                i++;
            }
            while( resourceProject[i] != '\n' ) {
                printf("%c",resourceProject[i++]);
            }
            printf("  )\n");
        } 


        //删除单行注释
        if ( resourceProject[i] == '/' && resourceProject[i + 1] == '/') {
            while( resourceProject[i] != '\n' ) {
                i++;
            }
        }

        //删除多行注释
        if ( resourceProject[i] == '/' && resourceProject[i + 1] == '*') {
            i = i + 2;
            while( resourceProject[i] != '*' && resourceProject[i + 1] != '/') {
                i++;
                if( i == end ) {
                    break;
                }
            }
            i = i + 2;
        }

        //删除换行符 制表符
        if ( resourceProject[i] != '\n' && resourceProject[i] != '\t') {
            temp[count] = resourceProject[i];
            count++;
        }
    }
    temp[count] ='$';//末位增加结束符 
    strcpy( resourceProject , temp );
    return count;
}

/*****查找保留字 1~32*****/
int searchReserve(char reserveWord[][10], char s[]) {
    int i;
    for ( i = 0; i < 32; i++ ) {
        if (strcmp(reserveWord[i], s) == 0) {
            i++;
            return i;
        }
    }
    return -1;

}

/*****判断是否为数字*****/
int IsDigit(char digit) {
    if (digit >= '0'&&digit <= '9') {
        return 1;
    }else {
        return 0;
    }
}

/*****判断是否为字母*****/
int IsLetter(char letter) {
    if ( (letter >= 'a'&&letter <= 'z') || (letter >= 'A'&&letter <= 'Z') || letter=='_') {
        return 1;
    }else {
       return 0;
    }
}

int Scaning(char resourceProject[] , char token[]) {
    int i, count = 0;
    char ch;
    ch = resourceProject[pProject];

    //过滤空格
    while (ch ==' ') {
        pProject++;
        ch = resourceProject[pProject];
    }

    //每次收集前先清零
    for (i = 0; i<20; i++) {
        token[i] = '\0';

    }

    /*****判断运算符 33~54*****/
    //加+ 减- 乘* 除/ 求余% 小于< 大于> 赋值= 非! 位与& 位或| 位异或^ 33~44
    if (ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '%' || 
        ch == '<' || ch == '>' || ch == '=' || ch == '!' || ch == '&' ||
        ch == '|' || ch == '^') {
        token[0] = resourceProject[pProject];
        token[1] = '\0';
        for (i = 0; i<22; i++) {
                if (strcmp(token, operatorWord[i]) == 0) {
                    t = 33 + i;
                    break;
                }
            }
        pProject++;
        return t;
    }else if (resourceProject[pProject] == '+') {
        pProject++;
        if (resourceProject[pProject] == '+') {         //自增++ 45
            t = 45;
        }
        pProject++;
        return t;
    }else if (resourceProject[pProject] == '-') {
        pProject++;
        if (resourceProject[pProject] == '-') {         //自减-- 46
            t = 46;
        }
        pProject++;
        return t;
    }else if (resourceProject[pProject] == '<') {
        pProject++;
        if (resourceProject[pProject] == '=') {         //小于等于<= 47
            t = 47;
        }else if (resourceProject[pProject] == '<') {   //左移<< 48
            pProject--;
            t = 48;
        }
        pProject++;
        return t;
    }else  if (resourceProject[pProject] == '>') {
        pProject++;
        if (resourceProject[pProject] == '=') {         //大于等于>= 49
            t = 49;
        }else if (resourceProject[pProject] == '>') {   //右移>> 50
            t = 50;
        }
        pProject++;
        return t;
    }else  if (resourceProject[pProject] == '=') {
        pProject++;
        if (resourceProject[pProject] == '=') {         //等于== 51
            t = 51;
        }
        pProject++;
        return t;
    }else  if (resourceProject[pProject] == '!') {
        pProject++;
        if (resourceProject[pProject] == '=') {         //不等于!= 52
            t = 52;
        }
        pProject++;
        return t;
    }else  if (resourceProject[pProject] == '&') {
        pProject++;
        if (resourceProject[pProject] == '&') {         //逻辑与&& 53
            t = 53;
        }
        pProject++;
        return t;
    } else  if (resourceProject[pProject] == '|') {
        pProject++;
        if (resourceProject[pProject] == '|') {         //逻辑或|| 54
            t = 54;
        }
        pProject++;
        return t;
    }

    /*****判断分隔符 55~70*****/
    else if (ch == '(' || ch == ')' || ch == '[' || ch == ']' || ch == '{' || 
        ch == '}' || ch == ';' || ch == ',' || ch == '\"' || ch == '\'' ||
        ch == '#' || ch == '~' || ch == '\\' || ch == '.' || ch == '\?' ||
        ch == ':') {
        token[0] = resourceProject[pProject];
        token[1] = '\0';
        for (i = 0; i<16; i++) {
                if (strcmp(token, boredrWord[i]) == 0) {
                    t = 55 + i;
                    break;
                }
            }
        pProject++;
        return t;
    }

    /*****判断字母*****/
    else if (IsLetter(resourceProject[pProject])) {
            token[count] = resourceProject[pProject];
            count++;
            pProject++;
            while (IsLetter(resourceProject[pProject]) || IsDigit(resourceProject[pProject])) {
                token[count++] = resourceProject[pProject];
                pProject++;
            }

            token[count] = '\0';
            t =  searchReserve(reserveWord, token);
            if (t == -1) {
                t = 100;
            }
        return t;
    }

    /*****判断常数*****/
    else if (IsDigit(resourceProject[pProject])) {
        while (IsDigit(resourceProject[pProject])||(resourceProject[pProject]=='.'&&IsDigit(resourceProject[pProject+1]))) {
            token[count] = resourceProject[pProject];
            count++;
            pProject++;
            t = 99;
        }
        token[count] = '\0';
        return t;
    }

    /*****判断结束符*****/
    else  if (resourceProject[pProject] == '$') {     
        t = 0;
        return t;
    }

    /*****判断错误*****/
    else {
        printf("错误:非法字符%c！\n",ch);
        pProject ++;
        return -1;
    }


}

int main() {
    //读取源程序
    char resourceProject[10000];
    char token[20] = {0};
    int i;
    int end = 0;
    FILE *fp;
    if ( ( fp = fopen ( "data.txt","r") ) == NULL ) {
        printf("未找到源程序文档！");
        return 0;
    }

    //将源程序读入resourceProject数组
    resourceProject[ pProject ] = fgetc(fp);
    while(resourceProject[ pProject ] != EOF ) {
        pProject++;
        resourceProject[ pProject ] = fgetc(fp);
    }
    fclose(fp);
    end = filtrationResource( resourceProject , pProject );
    pProject = 0;

    while (t != 0) {
        //扫描
        t = Scaning(resourceProject, token);
        if (t == 100) {
            for (i = 0; i<1000; i++) {
                if (strcmp(identifier[i], token) == 0) {
                    break;
                }
                if (strcmp(identifier[i], "") == 0) {
                    strcpy(identifier[i], token);
                    break;
                }
            }
            printf("<%s,%d,标识符>\n", token,t);
        }else if (t == 99) {
            printf("<%s,%d,常数>\n", token,t);
        }else if (t >= 1 && t <= 32) {
            printf("<%s,%d,保留字>\n", reserveWord[t - 1],t);
        }else if (t >= 33 && t <= 54) {
            printf("<%s,%d,运算符>\n", operatorWord[t - 33],t);
        }else if (t >= 55 && t <= 70) {
            printf("<%s,%d,分隔符>\n", boredrWord[t - 55],t);
        }

    }
    printf("扫描结束！");
    return 0;
}
